// Variabel Kalibrasi (Contoh: Panjang Kartu ATM = 85.6 mm) const KNOWN_WIDTH_MM = 85.6; function measureObject(contours, hierarchy) { // 1. Cari kontur terbesar (asumsi: ini adalah benda referensi/kartu) let refContour = contours.get(0); let rect = cv.minAreaRect(refContour); let pixelWidth = Math.max(rect.size.width, rect.size.height); // 2. Hitung Rasio: Pixel per Millimeter (PPM) let ppm = pixelWidth / KNOWN_WIDTH_MM; // 3. Ukur kontur lainnya for (let i = 1; i < contours.size(); ++i) { let cnt = contours.get(i); let area = cv.contourArea(cnt); if (area < 500) continue; // Abaikan noise kecil let box = cv.minAreaRect(cnt); let widthMM = box.size.width / ppm; let heightMM = box.size.height / ppm; console.log(`Benda ${i}: ${widthMM.toFixed(3)} mm x ${heightMM.toFixed(3)} mm`); // .toFixed(3) akan menampilkan 3 angka di belakang koma (0.001) } }
